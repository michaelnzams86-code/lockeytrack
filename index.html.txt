<script>
    let data = JSON.parse(localStorage.getItem('my_expenses')) || [];

    function saveExpense() {
        const entry = {
            id: Date.now(),
            title: document.getElementById('title').value,
            amount: parseFloat(document.getElementById('amount').value),
            category: document.getElementById('category').value,
            date: document.getElementById('date').value
        };

        if(!entry.title || !entry.amount || !entry.date) return alert("Fill everything!");

        data.push(entry);
        localStorage.setItem('my_expenses', JSON.stringify(data));
        updateUI();
    }

    function updateUI() {
        const history = document.getElementById('history');
        history.innerHTML = "";
        let totals = { d:0, w:0, m:0 };
        const now = new Date();

        data.sort((a,b) => new Date(b.date) - new Date(a.date)).forEach(ex => {
            const exDate = new Date(ex.date);
            
            // Calc logic
            if(exDate.toDateString() === now.toDateString()) totals.d += ex.amount;
            if((now - exDate) / (1000*60*60*24) <= 7) totals.w += ex.amount;
            if(exDate.getMonth() === now.getMonth()) totals.m += ex.amount;

            history.innerHTML += `
                <div class="item">
                    <div>
                        <strong>${ex.title}</strong><br>
                        <span class="category-tag">${ex.category}</span>
                    </div>
                    <span>$${ex.amount}</span>
                </div>`;
        });

        document.getElementById('d-total').innerText = `$${totals.d}`;
        document.getElementById('w-total').innerText = `$${totals.w}`;
        document.getElementById('m-total').innerText = `$${totals.m}`;
    }

    // Set default date to today
    document.getElementById('date').valueAsDate = new Date();
    updateUI();
</script>
