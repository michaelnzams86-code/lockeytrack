        <div class="stats">
            <div class="stat-box">Today<br><strong id="dayVal">$0</strong></div>
            <div class="stat-box">Month<br><strong id="monthVal">$0</strong></div>
        </div>
        <input type="text" id="title" placeholder="What did you buy?">
        <input type="number" id="amt" placeholder="Amount">
        <input type="date" id="date">
        <button onclick="addExp()">Add Expense</button>
        <div id="list"></div>
    </div>

    <script>
        let expenses = JSON.parse(localStorage.getItem('my_data')) || [];
        function addExp() {
            const t = document.getElementById('title').value;
            const a = parseFloat(document.getElementById('amt').value);
            const d = document.getElementById('date').value;
            if(!t || !a || !d) return alert("Fill all fields");
            expenses.unshift({t, a, d});
            localStorage.setItem('my_data', JSON.stringify(expenses));
            update();
        }
        function update() {
            const list = document.getElementById('list');
            list.innerHTML = "";
            let dTot=0, mTot=0;
            const now = new Date();
            expenses.forEach(x => {
                const xDate = new Date(x.d);
                if(xDate.toDateString() === now.toDateString()) dTot += x.a;
                if(xDate.getMonth() === now.getMonth()) mTot += x.a;
                list.innerHTML += `<div class="item"><span>${x.t}</span><strong>$${x.a}</strong></div>`;
            });
            document.getElementById('dayVal').innerText = "$" + dTot;
            document.getElementById('monthVal').innerText = "$" + mTot;
        }
        document.getElementById('date').valueAsDate = new Date();
        update();
    </script>
